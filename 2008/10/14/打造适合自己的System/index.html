function! AutoClose()
    :inoremap ( ()<esc>i
    :inoremap " ""<esc>i
    :inoremap ' ''<esc>i
    :inoremap { {}<esc>i
    :inoremap [ []<esc>i
    :inoremap ) <c-r>=ClosePair(')')<cr>
    :inoremap } <c-r>=ClosePair('}')<cr>
    :inoremap ] <c-r>=ClosePair(']')<cr>
endf

function! ClosePair(char)
    if getline('.')[col('.') - 1] == a:char
        return "<right>"
    else
        return a:char
    endif
endf
